\section{\label{sec:numerics}Numerical validation with explicit reference/prediction overlays}

All panels compare an exact reference reduced state to a model prediction at identical parameters.
The exact reference is always one of the following:
\begin{equation}
\rho_S^{\mathrm{ED}}(\beta)=\frac{\Tr_B e^{-\beta H_{\mathrm{tot}}}}{\Tr e^{-\beta H_{\mathrm{tot}}}},
\end{equation}
computed by finite-bath exact diagonalization, or an exact closed-form sector distribution
when the reduced state is diagonal in a conserved basis. The suite is implemented in
\texttt{simulations/src/run\_nested\_hs\_suite.py}; figures and claim checks are
\texttt{plot\_nested\_hs\_suite.py} and
\texttt{validate\_nested\_hs\_claims.py}.

\subsection{Commuting Gaussian sector (NHS-CG-1)}

We use a commuting qutrit system
\begin{equation}
H_S=\mathrm{diag}(0,0.7,1.6),\qquad f=\mathrm{diag}(0,1,2),
\end{equation}
with harmonic bath
\begin{equation}
H_B=\sum_k\omega_k(a_k^\dagger a_k+\tfrac12),\qquad H_I=f\otimes\sum_k c_k x_k,
\end{equation}
where $x_k=(a_k+a_k^\dagger)/\sqrt{2\omega_k}$. The commuting prediction is
\begin{equation}
\rho_S^{\mathrm{pred}}\propto e^{-\beta(H_S-\lambda f^2)},
\qquad \lambda=\sum_k\frac{c_k^2}{2\omega_k^2}.
\end{equation}

\subsection{Commuting non-Gaussian sector (NHS-CNG-1)}

For the many-body stress test we use magnetization sectors of an $N$-qubit model,
with exact sector weights
\begin{equation}
p(m)\propto g(m)\exp\left(k_2 m^2+k_4 m^4\right),
\end{equation}
where $m\in\{-N,-N+2,\dots,N\}$ and $g(m)$ is the binomial degeneracy. This provides
an exact non-Gaussian commuting benchmark. The HS estimator uses
$X\sim\mathcal N(0,2k_2)$ and $Y\sim\mathcal N(0,8k_4)$;
for $k_4<0$, contour-rotated sampling is used.

\subsection{Non-commuting Gaussian sector (NHS-NCG-1)}

We use a non-commuting qutrit model,
\begin{equation}
H_S=\mathrm{diag}(0,0.9,1.8),\qquad f=\lambda_1+0.45\lambda_6+0.2\lambda_4,
\end{equation}
with linearly coupled harmonic bath mode. The exact reference is
$\rho_S^{\mathrm{ED}}$. The prediction uses a Gaussian-field quadrature reconstruction with
matched second-cumulant scale,
\begin{equation}
\rho_S^{\mathrm{pred}}=\frac{\mathbb E_x\left[e^{-\beta(H_S+xf)}\right]}{\Tr\,\mathbb E_x\left[e^{-\beta(H_S+xf)}\right]}.
\end{equation}

\subsection{Non-commuting non-Gaussian sector (NHS-NCNG-1)}

For the hardest regime we use the same non-commuting qutrit with an exact static
spin-field trace-out,
\begin{equation}
\rho_S^{\mathrm{ED}}=\sum_B p(B)\,\frac{e^{-\beta(H_S+Bf)}}{\Tr e^{-\beta(H_S+Bf)}}.
\end{equation}
This reference is compared against K2-only and K2+K4 reconstructions.
